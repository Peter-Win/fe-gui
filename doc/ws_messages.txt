createApp: object - Create application
  common
    name: string
    description: string
    author: string
    license: string
  tech
    packageManager: 'Yarn' | 'NPM'
    bundler: 'WebPack'
    language: 'TypeScript' | 'JavaScript'
    transpiler: 'Babel' | 'TypeScript' | 'None'
    framework: 'None' | 'React'
    styleCss: bool
    styleLess: bool
